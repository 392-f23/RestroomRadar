function StarRating({ rating, reviewCount }) {
  const stars = [];

  for (let i = 1; i <= 5; i++) {
    if (i <= rating) {
      // full star
      stars.push(
        <svg
          key={i}
          xmlns="http://www.w3.org/2000/svg"
          width="16"
          height="16"
          fill="#c59b08"
          className="bi bi-star-fill"

        </svg>
      );

      i === Math.ceil(rating) &&
      rating % 1 >= 0.3 &&
      rating % 1 <= 0.99

        <svg
          key={i}
          xmlns="http://www.w3.org/2000/svg"

        </svg>
      );
    } else {
      // empty star
      stars.push(
        <svg
          key={i}
